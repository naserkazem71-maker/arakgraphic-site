    function render(productsList){
      const root = document.getElementById('products');
      root.innerHTML = '';
      productsList.forEach(p=>{
        const el = document.createElement('div');
        el.className='card';
        el.innerHTML = `
          <img src="${p.img}" alt="">
          <h3>${p.title}</h3>
          <p>${p.desc}</p>
          <div class="price">${p.price}</div>
          <div style="margin-top:8px;display:flex;gap:8px">
            <button onclick='openModal(${p.id})' style="padding:8px 10px;border-radius:8px;border:0;background:#eee">جزئیات</button>
            <button onclick='addToCart(${p.id})' class="btn">افزودن به سبد</button>
          </div>
        `;
        root.appendChild(el);
      });
    }

    function filterBy(cat){
      if(cat==='all') render(products);
      else render(products.filter(p=>p.category===cat));
    }

    function filterProducts(){
      const q = document.getElementById('search').value.trim();
      if(!q) return render(products);
      const res = products.filter(p=> (p.title+p.desc).toLowerCase().includes(q.toLowerCase()));
      render(res);
    }

    function openModal(id){
      const p = products.find(x=>x.id===id);
      document.getElementById('modalImg').src = p.img;
      document.getElementById('modalTitle').innerText = p.title;
      document.getElementById('modalDesc').innerText = p.desc;
      document.getElementById('modalPrice').innerText = p.price;
      const ul = document.getElementById('modalSpecs'); ul.innerHTML='';
      p.specs.forEach(s=>{ const li=document.createElement('li'); li.innerText=s; ul.appendChild(li)});
      document.getElementById('modal').classList.add('open');
    }
    function closeModal(){ document.getElementById('modal').classList.remove('open'); }
